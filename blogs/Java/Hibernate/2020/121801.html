# validation参数验证

# 注解

注解 说明
@NotNull 元素不能为空
@NotEmpty 用来校验字符串、集合、map、数组不能为null或空 字符串传入空格也不可以）（集合需至少包含一个元素）
@Size(max=1, min=5) 指定的字符串、集合、map、数组长度必须在指定的1和5内 允许元素为null，字符串允许为空格
@NotBlank 只用来校验字符串不能为null，空格也是被允许的 。校验字符串推荐使用@NotEmpty
@Length(min=1,max=5) 只用来校验字符串，长度必须在指定的1和5内 允许元素为null
@Min() 校验数字（包括integer short long int 等）的最小值，不支持小数即double和float 允许元素为null
@Max() 校验数字（包括integer short long int 等）的最小值，不支持小数即double和float 允许元素为null
@Range(min=1,max=5) 用来校验数字或字符串的大小必须在指定的1和5内, 字符串会转成数字进行比较, 如果不是数字校验不通过, 允许元素为null
@DecimalMin("1.01") 元素必须是一个数字，其值必须大于等于1.01, 支持浮点类型
@DecimalMax("10.01") 元素必须是一个数字，其值必须小于等于10.01, 支持浮点类型
@Digits(integer=5,fraction=3) 元素必须是一个数字, 其整数部分不超过5位数, 小数部分不超过3位数
@Email 只用来验证字符串是否满足Email格式
@Pattern() 正则表达式匹配，可用来校验年月日格式，是否包含特殊字符（regexp = "^1[3-9]\d{9}$")(国内验证手机号)
@Positive 元素为正数
@PositiveOrZero 元素为正数或者0
@Negative 元素为负数
@NegativeOrZero 元素为负数或者0
@Future 只用来验证Date是未来时间
@FutureOrPresent 只用来验证Date是未来时间或者当前时间
@Past 只用来验证Date是过去时间
@PastOrPresent 只用来验证Date是过去时间或者当前时间
@AssertTrue 元素必须为true
@AssertFalse 元素必须为false

# 常见分组注解常

注解 说明
Default 默认
One 指定启用
Insert 插入时启用
Update 修改时启用
Select 查询时启用
Delete 删除时启用

# 例子

import org.hibernate.validator.constraints.Length;

import javax.validation.constraints.NotBlank;
import javax.validation.constraints.NotNull;

public class Query {
	
    // 数据库ID自增。修改、查询、删除必须传，其他不需要传
    @NotNull(message = "id不为空", group = {Update.class, Select.class, Delete.class})
    private Integer id;
    
    // 名字不能为null、不能为""、只有空格字符串" "，且长度再1和10之间。只有插入的时候需要传
    @NotBlank(message = "name不能为空", group = Insert.class)
    @Length(min = 1, max = 10, message = "name长度范围再1和10之间")
    private String name;

    public Integer getId() {
        return id;
    }

    public void setId(Integer id) {
        this.id = id;
    }
    
    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }
}

import com.cn.validationdemo.entity.Query;
import org.springframework.format.annotation.DateTimeFormat;
import org.springframework.http.ResponseEntity;
import org.springframework.validation.annotation.Validated;
import org.springframework.web.bind.annotation.*;

import java.util.Date;

@RestController
public class QueryController {

    @PostMapping("add")
    public ResponseEntity<Object> add(@Validated({Default.class, Insert.class}) @RequestBody Query query) {
        return ResponseEntity.ok(query);
    }
    
    @DeleteMapping("remove")
    public ResponseEntity<Object> remove(@Validated({Default.class, Delete.class}) @RequestBody Query query) {
        return ResponseEntity.ok(query);
    }

    @PutMapping("edit")
    public ResponseEntity<Object> edit(@Validated({Default.class, Update.class}) @RequestBody Query query) {
        return ResponseEntity.ok(query);
    }
    
    @GetMapping("get-id")
    public ResponseEntity<Object> getById(@Validated({Default.class, Select.class}) @RequestBody Query query) {
        return ResponseEntity.ok(query);
    }
}
