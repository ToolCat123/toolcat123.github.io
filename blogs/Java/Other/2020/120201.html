# Apache FileUtils + FastJson

public class Test {
    public static void main(String[] args) throws IOException {
        String jsonString = FileUtils.readFileToString(new File("D:/IdeaProjects/github/hciot-v3/hciot-modules/hciot-modules-iot/src/main/resources/device-category.json"), "UTF-8");
        List<Data> list = JSON.parseArray(jsonString, Data.class);
        for (Data data1 : list) {
            if (data1.getParentId().equals(0)) {
                data1.setTreePath("/");
            } else {
                for (Data data2 : list) {
                    if (data1.getParentId().equals(data2.getId())) {
                        data1.setTreePath(data2.getTreePath() + data1.getId() + "/");
                        break;
                    }
                }
            }
            String sql = "INSERT INTO `hciot`.`prod_base_tree`(`id`, `parent_id`, `code`, `name`, `type`, `sort`, `tree_path`) VALUES (" + data1.getId() + ", " + data1.getParentId() + ", '" + data1.getKey() + "', '" + data1.getName() + "', 0, 0, '" + data1.getTreePath() + "');";
            FileUtils.writeLines(new File("C:/Users/admin/Desktop/" + System.currentTimeMillis() + ".sql"), Collections.singleton(sql), true);
        }
    }
}
